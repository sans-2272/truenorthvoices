function submitPoll() { const options = document.getElementsByName('identity'); let selectedValue = null; for (let option of options) { if (option.checked) { selectedValue = option.value; break; } } const resultDiv = document.getElementById('pollResult'); if (selectedValue) { resultDiv.innerHTML = `Thank you! You voted for: ${selectedValue}`; } else { resultDiv.innerHTML = `Please select an option before submitting.`; } }
